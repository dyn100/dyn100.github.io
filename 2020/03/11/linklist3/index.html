question1 插入

tips：为了加快遍历速度。采用了二分查找

#include<iostream>
using namespace std;

int search(int n,int x,int *p) //二分查找，确定插入位置 
{
    int l=0,r=n-1,mid=(l+r)/2;
    if(p[r]<x)
        return r+1; 
    while(l<r)
    {
        if(x<p[mid])
        {
            r=mid;
            mid=(l+r)/2;
        }
        else
        {
            l=mid+1;
            mid=(l+r)/2;
        }
    }
    return mid;
}


int main(){
    int a,b;
    cout<<"input length"<<endl;
    cin>>a; //输入数组长度
    int *p=new int[a+1];
    for(int i=0;i<a;i++){
        cin>>p[i] ;
        if(i>0){
            if(p[i]<=p[i-1]){
                cout<<"invalid!"<<endl; //如果数组不是递增排列，则显示非法 
                return 1;
            }
        }
    }  
    cout<<"input num u want insert"<<endl;
    cin>>b; //输入要插入的数 
    int pos=search(a,b,p);
    //cout<<pos;
    //重排数组 
    for(int i=a;i>pos;i--){
        p[i]=p[i-1];
    } 
    p[pos]=b;
    //输出 
    for(int i=0;i<=a;++i)
    {
        printf("%d ",p[i]);
    }
    return 0;
}

void del(linklist &p,int mink,int maxk) {
    linklist a,b;
    a = p;
    b = p;
    while (a->next->data < mink) {
        a = a->next;
        b = b->next;
    }
    while (b->next->data <= maxk ) {
        b = b->next;
        if (b->next == NULL) {
            break;
        }
    }
    a->next = b->next;

}

int main() {
    linklist p;
    create(p);
    int mink, maxk;
    cout << "input min and max" << endl;
    cin >> mink >> maxk;
    del(p, mink, maxk);
    cout << "answer：" << endl;;

question2 循环链表就地逆置

tips：和单链表逆置比起来可能更简单，毕竟不需要判断位置(循环长度)

#include <iostream>
#include <stdio.h>  
#include <stdlib.h> 
using namespace std;

typedef struct node {
    int data;
    node* next;
}node, * linklist;

//创建链表
int create(linklist &p) {
    p = new node;
    int len,num;
    linklist tmp, tmp2;
    p->next = NULL;
    tmp = p;
    cout << "length" << endl; //输入链表长度
    cin >> len;
    cout << "input data" << endl;;
    for (int i = 0; i < len; i++) {
        cin >> num;  //输入数据
        if (i == 0) {
            p->data = num;
            continue;
        }
        tmp2 = new node;
        tmp2->data = num;
        tmp2->next = NULL;
        tmp->next = tmp2;
        tmp = tmp->next;
    }
    tmp->next = p;
    return len;
}


//逆置链表
void reverse(linklist &p,int len) {
    linklist a, b,c;
    //判断链表是否非法
    if (p == NULL || p->next == NULL) {
        cout << "invalid!" << endl;
        return;
    }
    a = p;
    b = p->next;
    c = p->next->next;
    a->next = NULL;
    for (int i = 0; i < len; i++) {
        b->next = a;
        a = b;
        b = c;
        if (i == len - 1) {
            continue;
        }
        c = c->next;        
    }

}

int main() {
    linklist p;
    int num=create(p);
    reverse(p, num);
    cout << "answer：" << endl;;
    //打印链表
    for(int i=0;i<num;i++) {
        cout << p->data<<" ";    
        p = p->next;
    }
}

    for (int i = 0; i < len; i++) {
        cin >> num >> num2;  //输入数据

//计算,参数为链表和变量
int calc(linklist& p, int num) {
    linklist tmp = p;
    int ans = 0;
    while (tmp->next != NULL) {
        tmp = tmp->next;
        ans = ans + tmp->data * (pow(num, tmp->factorial));

    }

    return ans;

}

int main() {
    linklist p;
    int num, sum;
    create(p);
    cout << "imput data" << endl;
    cin >> num;
    sum = calc(p, num);
    cout << "answer：" << sum << endl;
